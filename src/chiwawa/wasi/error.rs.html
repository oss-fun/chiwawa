1//! WASI error codes and handling.
2//!
3//! This module defines WASI Preview 1 error codes as the [`WasiError`] enum,
4//! with conversions to and from errno values.
5
6use thiserror::Error;

8/// Result type for WASI operations.
9pub type WasiResult<T> = Result<T, WasiError>;

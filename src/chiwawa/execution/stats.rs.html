8/// Collects per-instruction execution counts.
9#[derive(Debug)]
10pub struct ExecutionStats {
11    total_instructions: AtomicU64,
12    per_instruction: FxHashMap<usize, AtomicU64>,
13}
14
15impl ExecutionStats {
16    /// Creates a new statistics collector.
17    pub fn new() -> Self {
18        Self {
19            total_instructions: AtomicU64::new(0),
20            per_instruction: FxHashMap::default(),
21        }
22    }
23
24    /// Records execution of a single instruction by handler index.
25    pub fn record_instruction(&mut self, handler_index: usize) {
26        self.total_instructions.fetch_add(1, Ordering::Relaxed);
27        self.per_instruction
28            .entry(handler_index)
29            .or_insert_with(|| AtomicU64::new(0))
30            .fetch_add(1, Ordering::Relaxed);
31    }

271    /// Prints statistics summary to stderr.
272    pub fn report(&self) {
273        let total = self.total_instructions.load(Ordering::Relaxed);
274
275        eprintln!("=== Execution Statistics ===");
276        eprintln!("Total instructions executed: {}", total);
277
278        if total == 0 {
279            eprintln!("=======================");
280            return;
281        }
282
283        // Collect and sort instruction counts
284        let mut counts: Vec<(usize, u64)> = self
285            .per_instruction
286            .iter()
287            .map(|(idx, count)| (*idx, count.load(Ordering::Relaxed)))
288            .collect();
289
290        counts.sort_by(|a, b| b.1.cmp(&a.1));
291
292        eprintln!("\nTop Instructions:");
293        let top_n = 20.min(counts.len());
294        for (idx, count) in counts.iter().take(top_n) {
295            let name = Self::get_instruction_name(*idx);
296            let percentage = (*count as f64 / total as f64) * 100.0;
297            eprintln!(
298                "  {:25} {:12} ({:5.1}%)",
299                format!("{}:", name),
300                count,
301                percentage
302            );
303        }
304
305        if counts.len() > top_n {
306            eprintln!("  ... and {} more", counts.len() - top_n);
307        }
308
309        eprintln!("=======================");
310    }

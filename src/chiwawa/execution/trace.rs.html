10/// Event types that can trigger tracing.
11#[derive(Debug, Clone, PartialEq)]
12pub enum TraceEvent {
13    All,
14    Store,
15    Load,
16    Call,
17    Branch,
18}
19
20impl TraceEvent {
21    /// Parses trace event from string (case-insensitive).
22    pub fn from_str(s: &str) -> Option<Self> {
23        match s.to_lowercase().as_str() {
24            "all" => Some(TraceEvent::All),
25            "store" => Some(TraceEvent::Store),
26            "load" => Some(TraceEvent::Load),
27            "call" => Some(TraceEvent::Call),
28            "branch" => Some(TraceEvent::Branch),
29            _ => None,
30        }
31    }
32}
33
34/// Resources to include in trace output.
35#[derive(Debug, Clone, PartialEq)]
36pub enum TraceResource {
37    PC,
38    Stack,
39    Memory,
40    Locals,
41    Globals,
42}
43
44impl TraceResource {
45    /// Parses trace resource from string (case-insensitive).
46    pub fn from_str(s: &str) -> Option<Self> {
47        match s.to_lowercase().as_str() {
48            "pc" => Some(TraceResource::PC),
49            "stack" => Some(TraceResource::Stack),
50            "memory" => Some(TraceResource::Memory),
51            "locals" => Some(TraceResource::Locals),
52            "globals" => Some(TraceResource::Globals),
53            _ => None,
54        }
55    }
56}

58/// Configuration for execution tracing.
59#[derive(Debug, Clone)]
60pub struct TraceConfig {
61    pub events: Vec<TraceEvent>,
62    pub resources: Vec<TraceResource>,
63    pub output_path: Option<String>,
64}
65
66impl TraceConfig {
67    /// Creates trace configuration from optional string arguments.
68    pub fn new(
69        events: Option<Vec<String>>,
70        resources: Option<Vec<String>>,
71        output_path: Option<String>,
72    ) -> Self {
73        let events = if let Some(event_strs) = events {
74            event_strs
75                .iter()
76                .filter_map(|s| TraceEvent::from_str(s))
77                .collect()
78        } else {
79            vec![TraceEvent::All]
80        };
81
82        let resources = if let Some(resource_strs) = resources {
83            resource_strs
84                .iter()
85                .filter_map(|s| TraceResource::from_str(s))
86                .collect()
87        } else {
88            vec![
89                TraceResource::PC,
90                TraceResource::Stack,
91                TraceResource::Locals,
92                TraceResource::Globals,
93            ]
94        };
95
96        Self {
97            events,
98            resources,
99            output_path,
100        }
101    }
102
103    /// Returns true if the given instruction should be traced.
104    pub fn should_trace_event(&self, handler_index: usize) -> bool {
105        if self.events.contains(&TraceEvent::All) {
106            return true;
107        }
108
109        // Check if the instruction matches any trace event
110        for event in &self.events {
111            match event {
112                TraceEvent::Store => {
113                    if (HANDLER_IDX_I32_STORE..=HANDLER_IDX_I64_STORE32).contains(&handler_index) {
114                        return true;
115                    }
116                }
117                TraceEvent::Load => {
118                    if (HANDLER_IDX_I32_LOAD..=HANDLER_IDX_I64_LOAD32_U).contains(&handler_index) {
119                        return true;
120                    }
121                }
122                TraceEvent::Call => {
123                    if handler_index == HANDLER_IDX_CALL
124                        || handler_index == HANDLER_IDX_CALL_INDIRECT
125                    {
126                        return true;
127                    }
128                }
129                TraceEvent::Branch => {
130                    if handler_index == HANDLER_IDX_BR
131                        || handler_index == HANDLER_IDX_BR_IF
132                        || handler_index == HANDLER_IDX_BR_TABLE
133                    {
134                        return true;
135                    }
136                }
137                TraceEvent::All => return true,
138            }
139        }
140
141        false
142    }
143}

145/// Writes execution traces to file or stderr.
146pub struct Tracer {
147    config: TraceConfig,
148    output: Box<dyn Write>,
149}
150
151impl Tracer {
152    /// Creates a new tracer with the given configuration.
153    pub fn new(config: TraceConfig) -> io::Result<Self> {
154        let output: Box<dyn Write> = if let Some(ref path) = config.output_path {
155            Box::new(File::create(Path::new(path))?)
156        } else {
157            Box::new(io::stderr())
158        };
159
160        Ok(Self { config, output })
161    }
162
163    /// Records a single instruction execution to the trace output.
164    pub fn trace_instruction(
165        &mut self,
166        ip: usize,
167        handler_index: usize,
168        value_stack: &[Val],
169        locals: &[Val],
170        global_addrs: &[super::global::GlobalAddr],
171    ) {
172        if !self.config.should_trace_event(handler_index) {
173            return;
174        }
175
176        let mut parts = Vec::new();
177
178        // PC (instruction pointer)
179        if self.config.resources.contains(&TraceResource::PC) {
180            parts.push(format!("PC:{:04}", ip));
181        }
182
183        // Instruction name
184        let instr_name = Self::get_instruction_name(handler_index);
185        parts.push(format!("Instr:{}", instr_name));
186
187        // Stack
188        if self.config.resources.contains(&TraceResource::Stack) {
189            let stack_str = self.format_stack(value_stack);
190            parts.push(format!("Stack:{}", stack_str));
191        }
192
193        // Locals
194        if self.config.resources.contains(&TraceResource::Locals) {
195            let locals_str = self.format_locals(locals);
196            parts.push(format!("Locals:{}", locals_str));
197        }
198
199        // Globals
200        if self.config.resources.contains(&TraceResource::Globals) {
201            let globals_str = self.format_globals(global_addrs);
202            parts.push(format!("Globals:{}", globals_str));
203        }
204
205        // Write trace line
206        let trace_line = format!("[{}]\n", parts.join(" | "));
207        let _ = self.output.write_all(trace_line.as_bytes());
208        let _ = self.output.flush();
209    }
210
211    fn format_globals(&self, global_addrs: &[super::global::GlobalAddr]) -> String {
212        if global_addrs.is_empty() {
213            return "[]".to_string();
214        }
215
216        let values: Vec<String> = global_addrs
217            .iter()
218            .map(|g| Self::format_val(&g.get()))
219            .collect();
220        format!("[{}]", values.join(","))
221    }
222
223    fn format_stack(&self, stack: &[Val]) -> String {
224        if stack.is_empty() {
225            return "[]".to_string();
226        }
227
228        let values: Vec<String> = stack.iter().map(|v| Self::format_val(v)).collect();
229        format!("[{}]", values.join(","))
230    }
231
232    fn format_locals(&self, locals: &[Val]) -> String {
233        if locals.is_empty() {
234            return "[]".to_string();
235        }
236
237        let values: Vec<String> = locals.iter().map(|v| Self::format_val(v)).collect();
238        format!("[{}]", values.join(","))
239    }
240
241    fn format_val(val: &Val) -> String {
242        match val {
243            Val::Num(num) => match num {
244                super::value::Num::I32(v) => format!("I32({})", v),
245                super::value::Num::I64(v) => format!("I64({})", v),
246                super::value::Num::F32(v) => format!("F32({})", v),
247                super::value::Num::F64(v) => format!("F64({})", v),
248            },
249            Val::Vec_(vec) => match vec {
250                super::value::Vec_::V128(v) => format!("V128({})", v),
251            },
252            Val::Ref(r) => match r {
253                super::value::Ref::RefNull => "RefNull".to_string(),
254                super::value::Ref::FuncAddr(_) => "FuncAddr".to_string(),
255                super::value::Ref::RefExtern(_) => "RefExtern".to_string(),
256            },
257        }
258    }
